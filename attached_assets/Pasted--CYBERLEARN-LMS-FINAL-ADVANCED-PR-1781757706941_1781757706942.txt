## ROLE & CONTEXT

You are a senior full-stack desktop application engineer. You are building
a 100% offline, hardware-locked, cybersecurity Learning Management System
(LMS) desktop application. This software is inspired by Udemy + Coursera +
Pluralsight but runs entirely offline — NO internet connection is ever made,
NO CDN is used, NO telemetry, NO remote calls of any kind. Everything runs
locally on the machine.

The app targets two platforms only:
- Kali Linux (x64)
- Raspberry Pi OS (armv7l and arm64)

Packaged as .deb and .AppImage via electron-builder.

## ABSOLUTE RULES — NEVER VIOLATE THESE

1. ZERO network calls. No fetch(), no axios, no http, no https to any
   external URL. Ever. All assets, fonts, libraries are bundled locally.

2. nodeIntegration is ALWAYS false in all BrowserWindows.

3. contextIsolation is ALWAYS true in all BrowserWindows.

4. All Node.js access from renderer goes through IPC + preload bridge ONLY.

5. DevTools are ALWAYS disabled in production builds.

6. setContentProtection(true) on EVERY BrowserWindow — no exceptions.

7. Every single database write to sensitive fields uses AES-256-GCM
   encryption before INSERT. Every read decrypts after SELECT.

8. No hardcoded secrets, keys, or passwords anywhere in source code.

9. All content files (videos, PDFs) served via custom Electron protocol
   only — never expose raw filesystem paths to renderer.

10. AppArmor profile blocks all network at OS level — deny inet, deny inet6.

## COMPLETE FEATURE SPECIFICATION

### FEATURE 1 — HWID LICENSE SYSTEM

File: `main/security/hwid.js`

```javascript
const { machineIdSync } = require('node-machine-id');
const crypto = require('crypto');
const db = require('../db/connection');
const { encrypt, decrypt, deriveKey } = require('./crypto');

function getHWID() {
  const raw = machineIdSync({ original: true });
  return crypto.createHash('sha256').update(raw).digest('hex');
}

function checkLicense() {
  const hwid = getHWID();
  const row = db.prepare('SELECT * FROM hwid_license WHERE hwid_hash = ?').get(hwid);
  if (!row || !row.is_valid) return { valid: false, reason: 'NO_LICENSE' };
  return { valid: true };
}

function activateLicense(licenseKey) {
  const hwid = getHWID();
  // Validate key format: XXXX-XXXX-XXXX-XXXX
  const pattern = /^[A-Z0-9]{4}-[A-Z0-9]{4}-[A-Z0-9]{4}-[A-Z0-9]{4}$/;
  if (!pattern.test(licenseKey)) return { success: false, error: 'INVALID_FORMAT' };
  // Check not already used on another HWID
  const existing = db.prepare('SELECT * FROM hwid_license WHERE license_key_enc = ?')
    .get(encrypt(licenseKey, deriveKey(hwid)));
  if (existing) return { success: false, error: 'ALREADY_USED' };
  db.prepare(`INSERT INTO hwid_license (hwid_hash, license_key_enc, activated_at, is_valid)
              VALUES (?, ?, ?, 1)`)
    .run(hwid, encrypt(licenseKey, deriveKey(hwid)), Date.now());
  return { success: true };
}

module.exports = { getHWID, checkLicense, activateLicense };
```

### FEATURE 2 — AES-256-GCM ENCRYPTION

File: `main/security/crypto.js`

```javascript
const crypto = require('crypto');

// Derive 32-byte key from HWID using PBKDF2 — never stored on disk
function deriveKey(hwid) {
  const salt = Buffer.from('cyberlearn-salt-v1', 'utf8');
  return crypto.pbkdf2Sync(hwid, salt, 100000, 32, 'sha256');
}

function encrypt(plaintext, key) {
  const iv = crypto.randomBytes(12);
  const cipher = crypto.createCipheriv('aes-256-gcm', key, iv);
  const encrypted = Buffer.concat([cipher.update(String(plaintext), 'utf8'), cipher.final()]);
  const tag = cipher.getAuthTag();
  return `${iv.toString('hex')}:${tag.toString('hex')}:${encrypted.toString('hex')}`;
}

function decrypt(payload, key) {
  const [ivHex, tagHex, encHex] = payload.split(':');
  const decipher = crypto.createDecipheriv('aes-256-gcm', key, Buffer.from(ivHex, 'hex'));
  decipher.setAuthTag(Buffer.from(tagHex, 'hex'));
  return decipher.update(Buffer.from(encHex, 'hex')).toString('utf8') + decipher.final('utf8');
}

module.exports = { deriveKey, encrypt, decrypt };
```

### S4 — Login Rate Limit + 15-Minute Lockout
```javascript
// Before checking password:
if (user.lockout_until > Date.now()) {
  const remaining = Math.ceil((user.lockout_until - Date.now()) / 60000);
  return { error: 'LOCKED', minutesRemaining: remaining };
}
// After wrong password:
const newFails = user.failed_logins + 1;
if (newFails >= 5) {
  db.prepare('UPDATE users SET failed_logins=?, lockout_until=? WHERE id=?')
    .run(newFails, Date.now() + 15 * 60 * 1000, user.id);
} else {
  db.prepare('UPDATE users SET failed_logins=? WHERE id=?').run(newFails, user.id);
}
// On successful login: reset failed_logins=0, lockout_until=0
```

### S6 — 30-Minute Idle Auto-Logout
```javascript
// renderer/src/hooks/useIdleTimer.js
export function useIdleTimer(onIdle) {
  useEffect(() => {
    let timer;
    const reset = () => {
      clearTimeout(timer);
      timer = setTimeout(onIdle, 30 * 60 * 1000);
    };
    const events = ['mousemove','keydown','mousedown','touchstart','scroll'];
    events.forEach(e => window.addEventListener(e, reset, { passive: true }));
    reset();
    return () => {
      clearTimeout(timer);
      events.forEach(e => window.removeEventListener(e, reset));
    };
  }, [onIdle]);
}
// onIdle → call 'auth:logout' IPC → clear tokens → redirect to /login
```

### S12 — Binary Integrity Check
```javascript
// main/security/integrity.js
const crypto = require('crypto');
const fs = require('fs');
const path = require('path');
const glob = require('glob');
const { encrypt, decrypt, deriveKey } = require('./crypto');

const INTEGRITY_FILE = path.join(app.getPath('userData'), 'integrity.enc');

function buildHashes(appPath) {
  return glob.sync(`${appPath}/**/*.js`, { ignore: '**/node_modules/**' })
    .map(f => ({ f, h: crypto.createHash('sha256').update(fs.readFileSync(f)).digest('hex') }));
}

function saveIntegrity(hwid) {
  const hashes = buildHashes(app.getAppPath());
  fs.writeFileSync(INTEGRITY_FILE, encrypt(JSON.stringify(hashes), deriveKey(hwid)));
}

function verifyIntegrity(hwid) {
  if (!fs.existsSync(INTEGRITY_FILE)) return false;
  const stored = JSON.parse(decrypt(fs.readFileSync(INTEGRITY_FILE,'utf8'), deriveKey(hwid)));
  const current = buildHashes(app.getAppPath());
  return stored.every((s,i) => current[i] && current[i].h === s.h);
}
```

```javascript
// Triggered after: lesson marked complete, quiz passed
// File: main/ipc/courses.js — checkCourseCompletion()

async function checkCourseCompletion(userId, courseId) {
  const allLessons = db.prepare(`
    SELECT l.id, l.type FROM lessons l
    JOIN modules m ON l.module_id = m.id
    WHERE m.course_id = ?`).all(courseId);

  const completedLessons = db.prepare(`
    SELECT lesson_id FROM progress
    WHERE user_id = ? AND completed = 1`).all(userId).map(r => r.lesson_id);

  const quizLessons = allLessons.filter(l => l.type === 'quiz');
  const allQuizzesPassed = quizLessons.every(l => {
    const quiz = db.prepare('SELECT id, pass_percent FROM quizzes WHERE lesson_id = ?').get(l.id);
    if (!quiz) return true;
    const best = db.prepare(`
      SELECT MAX(score * 100.0 / max_score) as pct FROM quiz_attempts
      WHERE user_id = ? AND quiz_id = ? AND passed = 1`).get(userId, quiz.id);
    return best && best.pct >= quiz.pass_percent;
  });

  const allLessonsComplete = allLessons.every(l => completedLessons.includes(l.id));

  if (allLessonsComplete && allQuizzesPassed) {
    await generateCertificate(userId, courseId);
    await checkAndAwardBadges(userId, courseId);
  }
}

async function generateCertificate(userId, courseId) {
  const existing = db.prepare('SELECT id FROM certificates WHERE user_id=? AND course_id=?')
    .get(userId, courseId);
  if (existing) return; // already issued

  const user   = db.prepare('SELECT username FROM users WHERE id=?').get(userId);
  const course = db.prepare('SELECT title FROM courses WHERE id=?').get(courseId);
  const settings = getSettings(); // platform_settings
  const ts = Date.now();
  const certHash = crypto.createHash('sha256')
    .update(`${userId}-${courseId}-${ts}`).digest('hex');

  // Generate PDF with pdfkit
  const pdfPath = path.join(contentDir, 'certificates', `${certHash}.pdf`);
  await buildCertificatePDF({
    studentName: user.username,
    courseName:  course.title,
    issueDate:   new Date(ts).toLocaleDateString('en-GB'),
    certHash,
    orgName:     settings.cert_org_name,
    sigName:     settings.cert_signature_name,
    logoPath:    settings.logo_path,
    outputPath:  pdfPath
  });

  db.prepare(`INSERT INTO certificates (user_id,course_id,issued_at,cert_hash,pdf_path)
              VALUES (?,?,?,?,?)`).run(userId, courseId, ts, certHash, pdfPath);
}
```

## ADVANCED FEATURE 3 — ADAPTIVE LEARNING ENGINE

Tracks every student's performance and dynamically recommends what to
study next. Fully local algorithm, no ML model needed.

### Implementation

```javascript
// main/adaptive/engine.js

class AdaptiveLearningEngine {

  // Score each lesson for a student based on:
  // - Quiz score history (weighted recent)
  // - Time spent vs expected duration (engagement)
  // - Number of retries
  // - Time since last access (spaced repetition)
  scoreLesson(userId, lessonId) {
    const progress = db.prepare(
      'SELECT * FROM progress WHERE user_id=? AND lesson_id=?').get(userId, lessonId);
    const attempts = db.prepare(
      `SELECT score, max_score, attempted_at FROM quiz_attempts
       WHERE user_id=? ORDER BY attempted_at DESC LIMIT 5`).all(userId);

    let score = 100;

    // Penalize low quiz scores
    if (attempts.length > 0) {
      const avgPct = attempts.reduce((a,b) => a + (b.score/b.max_score*100), 0) / attempts.length;
      if (avgPct < 50) score -= 40;
      else if (avgPct < 70) score -= 20;
    }

    // Spaced repetition: boost if not accessed in 3+ days
    if (progress && progress.last_accessed) {
      const daysSince = (Date.now() - progress.last_accessed) / 86400000;
      if (daysSince > 7) score += 50;
      else if (daysSince > 3) score += 25;
    }

    // Boost incomplete lessons
    if (!progress || !progress.completed) score += 30;

    return score;
  }

  // Returns ordered list of recommended lessons for student
  getRecommendations(userId, courseId, limit = 5) {
    const lessons = db.prepare(`
      SELECT l.id, l.title, l.type, m.title as module_title
      FROM lessons l JOIN modules m ON l.module_id = m.id
      WHERE m.course_id = ?`).all(courseId);

    return lessons
      .map(l => ({ ...l, priority: this.scoreLesson(userId, l.id) }))
      .sort((a, b) => b.priority - a.priority)
      .slice(0, limit);
  }

  // Weak area detection: topics where student consistently scores below 70%
  getWeakAreas(userId) {
    return db.prepare(`
      SELECT q.title, AVG(a.score * 100.0 / a.max_score) as avg_pct, COUNT(*) as attempts
      FROM quiz_attempts a
      JOIN quizzes q ON a.quiz_id = q.id
```

## ADVANCED FEATURE 8 — PROCTORED QUIZ MODE

Optional admin setting per quiz: enable proctored mode. Detects and logs
suspicious behavior during quiz.

### What It Monitors (all local, no camera)
```javascript
// main/security/proctor.js

class QuizProctor {
  constructor(userId, quizId) {
    this.events = [];
    this.userId = userId;
    this.quizId = quizId;
  }

  // 1. Window focus loss — student switched away from app
  // Detected via: mainWindow.on('blur')
  logFocusLoss(timestamp) {
    this.events.push({ type: 'FOCUS_LOSS', timestamp });
  }

  // 2. Unusual time gaps — student paused for >5 minutes mid-quiz
  logTimeGap(seconds) {
    if (seconds > 300) this.events.push({ type: 'TIME_GAP', seconds });
  }

  // 3. Rapid answer changes — answered then changed >3 times per question
  logAnswerChange(questionId, count) {
    if (count > 3) this.events.push({ type: 'ANSWER_CHANGES', questionId, count });
  }

  // 4. Quiz completed too fast — under 20% of expected time
  logSpeedFlag(expectedSec, actualSec) {
    if (actualSec < expectedSec * 0.2)
      this.events.push({ type: 'SPEED_FLAG', expectedSec, actualSec });
  }

  // Save all events to DB after quiz submission
  save() {
    db.prepare(`INSERT INTO quiz_proctoring
      (user_id, quiz_id, events_json, flagged, created_at)
      VALUES (?,?,?,?,?)`).run(
        this.userId, this.quizId,
        JSON.stringify(this.events),
        this.events.length > 0 ? 1 : 0,
        Date.now()
    );
  }
}
```

```javascript
// main/ipc/leaderboard.js

function getCourseLeaderboard(courseId, limit = 10) {
  return db.prepare(`
    SELECT
      u.username,
      e.completed_at,
      COUNT(DISTINCT p.lesson_id) as lessons_done,
      ROUND(AVG(qa.score * 100.0 / qa.max_score), 1) as avg_quiz_score,
      SUM(p.watched_seconds) / 60 as study_minutes,
      -- Rank score: completion + speed + quiz performance
      (
        CASE WHEN e.completed_at IS NOT NULL THEN 1000 ELSE 0 END +
        ROUND(AVG(qa.score * 100.0 / qa.max_score) * 5, 0) +
        (SUM(p.watched_seconds) / 60)
      ) as rank_score
    FROM enrollments e
    JOIN users u ON e.user_id = u.id
    LEFT JOIN progress p ON p.user_id = e.user_id
    LEFT JOIN lessons l ON p.lesson_id = l.id
    LEFT JOIN modules m ON l.module_id = m.id AND m.course_id = e.course_id
    LEFT JOIN quizzes qz ON qz.lesson_id = l.id
    LEFT JOIN quiz_attempts qa ON qa.quiz_id = qz.id AND qa.user_id = e.user_id
    WHERE e.course_id = ?
    GROUP BY e.user_id
    ORDER BY rank_score DESC
    LIMIT ?`).all(courseId, limit);
}
```

## ADVANCED FEATURE 11 — ADMIN OFFLINE BACKUP SCHEDULER

Automatic encrypted database + content backups on a schedule. No cloud.
Saves to a local folder chosen by admin.

### Implementation

```javascript
// main/backup/scheduler.js
const { CronJob } = require('cron'); // bundled

class BackupScheduler {
  constructor(settings) {
    this.settings = settings; // from platform_settings table
  }

  start() {
    // Admin configures: daily / weekly / on-close
    const schedule = this.settings.backup_schedule; // '0 2 * * *' = daily 2am

    if (schedule === 'on-close') {
      app.on('before-quit', () => this.runBackup());
      return;
    }

    this.job = new CronJob(schedule, () => this.runBackup(), null, true);
  }

  async runBackup() {
    const dest   = this.settings.backup_path; // user-chosen local folder
    const hwid   = getHWID();
    const ts     = new Date().toISOString().replace(/[:.]/g, '-');
    const dbPath = path.join(app.getPath('userData'), 'data', 'cyberlearn.db');

    // 1. Encrypt SQLite file with AES-256-GCM
    const encDb = encryptFile(fs.readFileSync(dbPath), deriveKey(hwid));
    fs.writeFileSync(path.join(dest, `backup-${ts}.cyberdb`), encDb);

    // 2. Optionally backup content folder (videos/PDFs) — optional, large
    if (this.settings.backup_include_content === '1') {
      // tar + encrypt content/
    }

    // 3. Log backup event
    writeAuditLog(null, 'AUTO_BACKUP', `Saved to ${dest}/backup-${ts}.cyberdb`);

    // 4. Rotate: keep only last N backups
    const maxBackups = parseInt(this.settings.backup_keep_count || '7');
    const existing = fs.readdirSync(dest)
      .filter(f => f.startsWith('backup-') && f.endsWith('.cyberdb'))
      .sort();
    while (existing.length > maxBackups) {
      fs.unlinkSync(path.join(dest, existing.shift()));
    }
  }
}
```

```javascript
// main/adaptive/studyPlan.js

function generateStudyPlan(courseId, hoursPerWeek, startDate) {
  const lessons = db.prepare(`
    SELECT l.id, l.title, l.type, l.duration_seconds, m.title as module
    FROM lessons l JOIN modules m ON l.module_id = m.id
    WHERE m.course_id = ? ORDER BY m.order_index, l.order_index`).all(courseId);

  const totalMinutes = lessons.reduce((sum, l) => {
    let mins = l.duration_seconds / 60;
    if (l.type === 'quiz') mins = 15;     // estimate 15 min per quiz
    if (l.type === 'terminal') mins = 30; // estimate 30 min per lab
    return sum + mins;
  }, 0);

  const minutesPerWeek = hoursPerWeek * 60;
  const weeksNeeded    = Math.ceil(totalMinutes / minutesPerWeek);

  // Distribute lessons across weeks respecting module boundaries
  const weeks = [];
  let weekLessons = [], weekMins = 0, weekNum = 1;

  for (const lesson of lessons) {
    const mins = lesson.duration_seconds / 60 || 20;
    if (weekMins + mins > minutesPerWeek && weekLessons.length > 0) {
      weeks.push({ week: weekNum++, lessons: weekLessons, estimatedMins: weekMins });
      weekLessons = []; weekMins = 0;
    }
    weekLessons.push(lesson);
    weekMins += mins;
  }
  if (weekLessons.length) weeks.push({ week: weekNum, lessons: weekLessons, estimatedMins: weekMins });

  return { totalMinutes, weeksNeeded, weeks, startDate };
}
```

## ADVANCED FEATURE 14 — CONTENT INTEGRITY VERIFICATION

Every video and PDF file has a SHA-256 hash stored at upload time.
On every access, the file is verified before serving to student.
Detects tampering, corruption, or unauthorized file replacement.

### Implementation

```javascript
// main/security/contentIntegrity.js

// At upload time (admin uploads file):
function hashAndStoreContent(filePath, lessonId) {
  const hash = crypto.createHash('sha256')
    .update(fs.readFileSync(filePath)).digest('hex');
  db.prepare('UPDATE lessons SET content_hash=? WHERE id=?').run(hash, lessonId);
  return hash;
}

// Before serving file via cyberlearn:// protocol:
function verifyContentFile(filePath, expectedHash) {
  const actual = crypto.createHash('sha256')
    .update(fs.readFileSync(filePath)).digest('hex');
  if (actual !== expectedHash) {
    writeAuditLog(null, 'CONTENT_TAMPER_DETECTED', filePath);
    throw new Error('FILE_INTEGRITY_VIOLATION');
  }
  return true;
}

// Database addition: content_hash TEXT column in lessons table
// Protocol handler checks hash before streaming any file
```
