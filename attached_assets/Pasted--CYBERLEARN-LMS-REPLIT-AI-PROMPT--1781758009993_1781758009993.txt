## WHAT TO BUILD

Build a full-stack web application called CyberLearn LMS —
an offline-capable cybersecurity Learning Management System.
This runs entirely in the browser as a web app.
Backend: Node.js + Express. Database: PostgreSQL (Replit built-in).
Frontend: React + Tailwind CSS.

---

## BACKEND MIDDLEWARE

### authMiddleware.js
```javascript
// Verify JWT access token on every protected route
// Attach req.user = { id, username, role }
// Return 401 if token missing/expired
// Return 403 if role check fails

export function requireAuth(req, res, next) {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    req.user = jwt.verify(token, process.env.ACCESS_TOKEN_SECRET);
    next();
  } catch {
    res.status(401).json({ error: 'Invalid token' });
  }
}

export function requireAdmin(req, res, next) {
  if (req.user?.role !== 'ADMIN')
    return res.status(403).json({ error: 'Admin only' });
  next();
}
```

// Track failed logins per username in DB

        x: 5 + Math.random() * 65,

      }}

      {username}
